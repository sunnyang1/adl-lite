This appendix provides proof sketches for the seven core theorems of Section~\ref{subsec:formal-semantics}, plus an optional CRDT property (Theorem~\ref{thm:crdt-convergence-appendix}) scoped to Phase~3. We recall the key definitions. The event alphabet is $\Sigma = \Sigma_{\text{life}} \cup \Sigma_{\text{comm}}$ with $\Sigma_{\text{life}} = \{\textsc{Register}, \textsc{Validate}, \textsc{Deprecate}, \textsc{Fork}, \textsc{Archive}\}$. A well-formed chain $C$ satisfies genesis anchoring, shared \textit{concept\_id}, cryptographic linkage, hash correctness, and pairwise distinct \textit{event\_id} values. The status map $f: \Sigma_{\text{life}} \to S$ is defined by $f(\textsc{Register}) = \textit{provisional}$, $f(\textsc{Validate}) = \textit{validated}$, $f(\textsc{Deprecate}) = \textit{deprecated}$, $f(\textsc{Fork}) = \textit{forked}$, and $f(\textsc{Archive}) = \textit{archived}$.

%-------------------------------------------------------------------------------
\subsection{Theorem~\ref{thm:determinism}: Determinism of Status Derivation}
%-------------------------------------------------------------------------------

\begin{theorem}[Determinism]\label{thm:determinism}
For any well-formed chain $C$, the status $\delta(C)$ is unique and is the least upper bound (LUB) of all lifecycle events in $C$. Formally:
\begin{equation}
    \forall C, C' \; . \; \Phi(C) \land \Phi(C') \land \operatorname{lub}(C_{\text{life}}) = \operatorname{lub}(C'_{\text{life}}) \implies \delta(C) = \delta(C').
\end{equation}
where $\operatorname{lub}(C_{\text{life}}) = \max_{\prec}\{f(e.\textit{type}) \mid e \in C_{\text{life}}\}$ is the maximum over the status lattice. Moreover, $\delta(C)$ is computable in $O(|C_{\text{life}}|)$ time by a single scan.
\end{theorem}

\begin{proof}[Proof Sketch]
The lifecycle subsequence $C_{\text{life}} = \operatorname{filter}(C, \Sigma_{\text{life}})$ is totally ordered because $\Phi(C)$ enforces cryptographic linkage. If $C_{\text{life}}$ is empty, $\delta(C) = \textit{provisional}$ by definition. Otherwise, the LUB over the status lattice is the maximum element, which is unique because the lattice order is antisymmetric. Thus $\delta(C) = \max_{\prec}\{f(e.\textit{type}) \mid e \in C_{\text{life}}\}$ is uniquely determined. The complexity claim follows because filtering $C$ takes one linear scan and computing the max over $C_{\text{life}}$ is $O(|C_{\text{life}}|)$.
\end{proof}

\subsection{Theorem~\ref{thm:confluence}: Fork Determinism}
%-------------------------------------------------------------------------------

\begin{theorem}[Fork Determinism, No Merge]\label{thm:confluence}
Let $\operatorname{fork}(C) = (C_{\text{fork}}, C')$ where $C_{\text{fork}} = C \oplus [e_{\textsc{Fork}}]$ and $C' = [e'_{\textsc{Register}}]$. Let $s = \delta(C)$ be the parent's status before the fork. Then $\delta(C_{\text{fork}}) = \max(s, \textit{forked})$ (the LUB of the parent's previous status and \textsc{Fork}) and $\delta(C') = \textit{provisional}$, independently, because $C_{\text{fork}}$ and $C'$ are disjoint after the fork point.
\end{theorem}

\begin{proof}[Proof Sketch]
By the CRDT LUB semantics, $\delta(C_{\text{fork}}) = \max\{\delta(C), f(\textsc{Fork})\} = \max(s, \textit{forked})$. If $s = \textit{validated}$ (order 3) and $f(\textsc{Fork}) = \textit{forked}$ (order 2), then $\max(3, 2) = 3$ and the parent remains \textit{validated}. If $s = \textit{provisional}$ (order 1), then $\max(1, 2) = 2$ and the parent becomes \textit{forked}. For the child, $\delta(C') = f(\textsc{Register}) = \textit{provisional}$ because $C'$ contains exactly one event. Independence holds because $C_{\text{fork}}$ appends to $C$ while $C'$ starts afresh; the two derivations are independent per-chain.
\end{proof}

\subsection{Theorem~\ref{thm:monotonicity-status}: Monotonicity of Status Transitions}
%-------------------------------------------------------------------------------

\begin{theorem}[Monotonicity of Status Transitions]\label{thm:monotonicity-status}
Let $C$ be well-formed with $\delta(C) = s$, and let $e_{\text{new}}$ satisfy $\Phi(C \oplus [e_{\text{new}}])$. Let $s' = \delta(C \oplus [e_{\text{new}}])$. Then either (i) $e_{\text{new}}.\textit{type} \notin \Sigma_{\text{life}}$ and $s' = s$, or (ii) $e_{\text{new}}.\textit{type} \in \Sigma_{\text{life}}$ and $(s, s')$ is a valid edge in the status machine.
\end{theorem}

\begin{proof}[Proof Sketch]
By induction on $|C|$. Base case: $C = [e_{\textsc{Register}}]$, $\delta(C) = \textit{provisional}$; any new lifecycle event is permitted by the ActionExecutor precondition system only if the transition is valid. Inductive step: if $e_{\text{new}} \notin \Sigma_{\text{life}}$, the lifecycle subsequence is unchanged, so $s' = s$; if $e_{\text{new}} \in \Sigma_{\text{life}}$, then $s' = f(e_{\text{new}}.\textit{type})$, and the ActionExecutor precondition system permits this only when $(s, s')$ is a valid edge.
\end{proof}

\subsection{Theorem~\ref{thm:boundedness}: Boundedness of Confidence}
%-------------------------------------------------------------------------------

\begin{theorem}[Boundedness]\label{thm:boundedness}
For any well-formed chain $C$, $\gamma_{\text{agg}}(C) \in [0, 1]$.
\end{theorem}

\begin{proof}[Proof Sketch]
If $V = [e \in C \mid e.\textit{type} = \textsc{Validate}]$ is empty, $\gamma_{\text{agg}}(C) = 0.0$ by definition. Otherwise, each $\textsc{Validate}$ event carries confidence $\in [0, 1]$; the per-actor maxima $\phi(a, V)$ are therefore in $[0, 1]$, and their mean is a convex combination in $[0, 1]$. The $\max(0.5, \cdot)$ floor ensures $c_{\text{base}} \geq 0.5$, and the $\min(1.0, \cdot)$ cap ensures $\gamma_{\text{agg}}(C) \leq 1.0$. Duplicate validators use $\operatorname{argmax}$, which preserves the upper bound. Hence $\gamma_{\text{agg}}(C) \in [0, 1]$.
\end{proof}

\subsection{Theorem~\ref{thm:monotonicity-confidence}: Monotonicity of Confidence}
%-------------------------------------------------------------------------------

\begin{theorem}[Monotonicity of Confidence]\label{thm:monotonicity-confidence}
Let $C$ be well-formed and $e_{\text{new}}$ a $\textsc{Validate}$ event from actor $a$ with confidence $c \geq c_{\text{base}}$. Let $C' = C \oplus [e_{\text{new}}]$. Then $\gamma_{\text{agg}}(C') \geq \gamma_{\text{agg}}(C)$. If $\gamma_{\text{agg}}(C) < 1.0$ and $a$ is a new validator, strict inequality holds.
\end{theorem}

\begin{proof}[Proof Sketch]
\textbf{Case 1 (new validator).} $N'_{\text{vals}} = N_{\text{vals}} + 1$, and $c'_{\text{base}} = \max(0.5, (N_{\text{vals}} \cdot c_{\text{base}}^{\text{raw}} + c) / (N_{\text{vals}} + 1)) \geq c_{\text{base}}$. The bonus term increases by $0.05$, so $\gamma_{\text{agg}}(C') \geq \gamma_{\text{agg}}(C)$. If $\gamma_{\text{agg}}(C) < 1.0$, the cap is not binding and the bonus yields strict inequality. For $N_{\text{vals}} = 0$, $\gamma_{\text{agg}}(C') = \min(1.0, \max(0.5, c)) \geq 0.5 > 0 = \gamma_{\text{agg}}(C)$. \textbf{Case 2 (duplicate validator).} $\phi'(a, V') = \max(\phi(a, V), c) \geq \phi(a, V)$, so $c'_{\text{base}} \geq c_{\text{base}}$ with unchanged bonus. Hence $\gamma_{\text{agg}}(C') \geq \gamma_{\text{agg}}(C)$.
\end{proof}

\subsection{Theorem~\ref{thm:consistency}: Status--Confidence Consistency}
%-------------------------------------------------------------------------------

\begin{theorem}[Status--Confidence Consistency]\label{thm:consistency}
If $\delta(C) = \textit{validated}$, then $\gamma_{\text{agg}}(C) > 0$ (in fact, $\geq 0.5$).
\end{theorem}

\begin{proof}[Proof Sketch]
$\delta(C) = \textit{validated}$ implies that the LUB of all lifecycle events in $C$ is $\textit{validated}$, which requires at least one $\textsc{Validate}$ event in $C_{\text{life}}$ (since no other lifecycle event type maps to $\textit{validated}$). Thus $V \neq \emptyset$. Every $\textsc{Validate}$ event carries confidence $\geq 0.5$ by the ActionExecutor precondition system, hence $\phi(a, V) \geq 0.5$ for all actors. Therefore $c_{\text{base}} = \max(0.5, \text{mean}(\phi)) \geq 0.5$, and the bonus term is non-negative, yielding $\gamma_{\text{agg}}(C) = \min(1.0, c_{\text{base}} + \text{bonus}) \geq 0.5 > 0$.
\end{proof}

\subsection{Theorem~9: CRDT Convergence of EventChain Merge}
%-------------------------------------------------------------------------------

\begin{theorem}[CRDT Convergence]\label{thm:crdt-convergence-appendix}
For any well-formed concurrent branches $B_1, B_2$ from the same genesis, the LWW-Set merge $\text{Merge}(B_1, B_2) = B_1 \cup B_2$ (deduplicated by \textit{event\_id}) satisfies commutativity, associativity, and idempotence. Moreover, $\delta(\text{Merge}(B_1, B_2))$ is uniquely determined.
\end{theorem}

\begin{proof}[Proof Sketch]
Set union is commutative and associative; deduplication by \textit{event\_id} is symmetric and idempotent. For idempotence, $\text{Merge}(B_1, B_1) = B_1$ because $\Phi$ guarantees pairwise distinct \textit{event\_id} values within $B_1$. Determinism of $\delta$ after merge follows because the merged set contains all lifecycle events from both branches; the LUB over the status lattice is uniquely determined (Theorem~\ref{thm:determinism}), so $\delta$ is independent of merge order.
\end{proof}

\subsection{Corollary: Event-Level G-Set CRDT}
%-------------------------------------------------------------------------------

\begin{corollary}[Event-Level G-Set CRDT]\label{cor:gset-crdt}
Each ADL Lite event is immutable (hash-locked), so the EventChain $C$ as an append-only log is a grow-only set (G-Set) CRDT: the merge of two chains is the union of their event sets, with no concurrent updates to resolve.
\end{corollary}

\begin{proof}[Proof Sketch]
An event's hash $e.h$ covers all fields including \textit{event\_id} and payload; any modification changes the hash, breaking linkage. Hence events are write-once. The EventChain supports only $\text{append}$ ($C \oplus [e]$), so concurrent branches add disjoint events, and the merge is simply set union.
\end{proof}

\subsection{Theorem~\ref{thm:wf-preservation}: Well-Formedness Preservation}
%-------------------------------------------------------------------------------

\begin{theorem}[Well-Formedness Preservation]\label{thm:wf-preservation}
For any well-formed chain $C$ and any event $e_{\text{new}}$ satisfying the preconditions of its action type, the extended chain $C' = C \oplus [e_{\text{new}}]$ is well-formed.
\end{theorem}

\begin{proof}[Proof Sketch]
We verify each $\text{WF}_i$ axiom for $C'$:
\begin{enumerate}[label=(\arabic*)]
    \item $\text{WF}_1$: The genesis event is unchanged, so $e_1.h_{\text{prev}} = \text{genesis}$ and $e_1.e_{\text{prev}} = \text{null}$ still hold.
    \item $\text{WF}_2$: $e_{\text{new}}$ shares the same $\text{concept\_id}$ by construction (the append operation enforces this), so all events in $C'$ share the same $\text{concept\_id}$.
    \item $\text{WF}_3$: $e_{\text{new}}$ is appended with $e_{\text{new}}.h_{\text{prev}} = e_n.h$ and $e_{\text{new}}.e_{\text{prev}} = e_n.\text{event\_id}$, where $e_n$ is the previous last event. All prior linkages are unchanged.
    \item $\text{WF}_4$: $e_{\text{new}}.h$ is computed as $\text{SHA-256}(\text{Canon}(e_{\text{new}}))$ by construction. All existing hashes are unchanged.
    \item $\text{WF}_5$: $e_{\text{new}}.\text{event\_id}$ is generated as a fresh UUID, so it is distinct from all existing $\text{event\_id}$ values in $C$.
    \item $\text{WF}_6$: The ActionExecutor rejects events with empty actor fields as part of payload validation, so $e_{\text{new}}.a \neq \text{null}$ and $e_{\text{new}}.a \neq \text{""}$.
    \item $\text{WF}_7$: The append operation enforces $e_{\text{new}}.t \geq e_n.t$, preserving timestamp monotonicity.
    \item $\text{WF}_8$: The ActionExecutor checks payload schema conformance via Pydantic validation before appending, so $\text{SchemaValid}(e_{\text{new}}.\tau, e_{\text{new}}.p)$ holds.
\end{enumerate}
Thus $\text{WF}(C')$ holds.
\end{proof}

\subsection{Summary of Proof Dependencies}
%-------------------------------------------------------------------------------

The seven core theorems form a dependency graph: Theorem~\ref{thm:determinism} (foundational) $\to$ Theorem~\ref{thm:confluence} (fork) $\to$ Theorem~\ref{thm:monotonicity-status} (preconditions); Theorem~\ref{thm:boundedness} (independent) $\to$ Theorem~\ref{thm:monotonicity-confidence} (cap argument) $\to$ Theorem~\ref{thm:consistency} (status--confidence); and Theorem~\ref{thm:wf-preservation} (Well-Formedness Preservation). The optional CRDT property (Theorem~\ref{thm:crdt-convergence-appendix}) and the G-Set corollary depend on Theorem~\ref{thm:determinism} and $\Phi$ respectively. Together, these proofs establish that ADL Lite's derivation semantics is deterministic, fork-deterministic, transition-monotonic, bounded, confidence-monotonic, status--confidence consistent, and well-formedness-preserving.
